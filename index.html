    </button>


    <!-- Output Area -->

    <div

      id="output"

      class="mt-6 p-4 border border-gray-200 bg-gray-50 rounded-lg text-gray-800 whitespace-pre-line"

    ></div>

  </div>


  <!-- Script to call Netlify Function -->

  <script>

    async function generateLetter(prompt) {

      try {

        const res = await fetch('/.netlify/functions/generate-letter', {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify({ prompt })

        });


        const data = await res.json();


        if (!res.ok) {

          console.error('API returned error:', data);

          document.getElementById('output').innerText = '⚠️ Sorry, there was a problem generating the letter. Please try again.';

          return;

        }


        if (data.text) {

          document.getElementById('output').innerText = data.text;

        } else {

          document.getElementById('output').innerText = '⚠️ No letter content returned.';

        }


      } catch (err) {

        console.error('Fetch error:', err);

        document.getElementById('output').innerText = '⚠️ Failed to contact the server. Please try again.';

      }

    }


    document.getElementById('generate-button').addEventListener('click', () => {

      const prompt = document.getElementById('user-input').value.trim();

      if (prompt.length === 0) {

        document.getElementById('output').innerText = '⚠️ Please enter a message first.';

        return;

      }

      document.getElementById('output').innerText = '⏳ Generating your letter...';

      generateLetter(prompt);

    });

  </script>
